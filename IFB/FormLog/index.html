    Field Log

    Overview

    Sometimes when build a form you want to capture changes to fields as users fill them out creating a log for that specific form. This can be accomplished by creating a string field with a large max length property and then mapping it to a multiline textbox on your form. In the code blocks below I have control on my form with the control ID of mtxtLog that I will be using to hold my log information and is mapped to my log field.

    Two ways we can accomplish this is:

    1. Write the same log code in each control we wish to monitor for changes
    2. Create a function on the form that we can then use on the a control we want to log an action.

    Both of these methods are fired from the event change of the control.

    Note

    A controls value can change twice, this can happen when a when for example a control is mapped to a date field. Double entries can occur when the date is removed, one entry is for when the control is set to a blank value. The next entry happens when Encompass updates the mapped field to // for the blank date. Additional logic may be needed to account for changes like this.

Public Function addLog(ByVal logAs as string, ByVal triggerControl as string) as String
    'ARGs:
    'logAs - this is a string that will be use to identify changes in the log
    'triggerControl - this is the control ID of the control that is being logged


    'Get the current user ID
    Dim currUser as string = Loan.Session.GetCurrentUser().ID

    'Get the date time of change
    Dim changeDateTime as string = DateTime.Now.ToString()

    'Find control
    Dim foundControl = FindControl(triggerControl)

    'Create a log string to add to log
    Dim logString as string = logAs & ":" & vbcrlf & "New Value: " & foundControl.Value & " - changed by: " & currUser & " on " & changeDateTime

    'If log has not value then add value
    'Make sure to update mtxtLog with the control being used for logging
    If mtxtLog.value = String.Empty Then
        mtxtLog.text = logString
    'Else log already has a value and we need to add to it
    Else
        Dim orgText = mtxtLog.text
        mtxtLog.text = String.Empty
        mtxtLog.text = orgText + vbcrlf + vbcrlf + logString
    End If
End Function

//New class that holds custom methods to be used
private class CustomMethods
{
    public static void addLog(string logAs, Control triggerControl, MultilineTextBox logControl)
    {
        /*
        *ARGs:
        *logAs - this is a string that will be use to identify changes in the log
        *triggerControl - this is the control ID of the control that is being logged
        *logControl = this is the multiline textbox that is used for the log
        */

        //Get the current user ID
        string currUser = EncompassApplication.CurrentLoan.Session.GetCurrentUser().ID;

        //Get the date time of change
        string changeDateTime = DateTime.Now.ToString();


        //Getting Controls Value
        //Getting the type of control passed
        Type controlType = triggerControl.GetType();

        //Use reflection to get the value of the control
        var valueProp = controlType.GetProperty("Value");   
        var cValue = valueProp.GetValue(triggerControl).ToString();

        //Create a log string to add to log
        string logString = logAs + ":" + Environment.NewLine + "New Value: " + cValue + " - changed by: " + currUser + " on " + changeDateTime;



        //If log has no value then add log string
        //Make sure to update mtxtLog with the control beign used for logging
        if (logControl.Value == String.Empty)
        {
            logControl.Text = logString;
        }
        //Else log already has a value and we need to add to it
        else
        {
            string orgText = logControl.Text;
            logControl.Text = String.Empty;
            logControl.Text = orgText + Environment.NewLine + Environment.NewLine + logString;
        }
    }

//Do not add a } for new class as there is already one
//from the original load event
