    Print Document to eFolder

    Overview

    The below code is used in conjunction with a Print Auto Select business rule to print an attachment to the eFolder and place it in the correct document placeholder. If the document placeholder is not part of the eFolder it will attempt to create it. If the conversion of the document take to long the document will remain in the File Manager and the user will receive a message letting them know the attachment is in the File Manager.

    Print Auto Select Rule

    Below is an example of how a Print Auto Select rule should be setup to work with this code:

    'Get the current User
Dim currUser as User = Loan.Session.GetCurrentUser()
'Create an array of Persona names
Dim arrPersonas() as string = {"Super Administrator", "Administrator"}
'Create boolean to hold if user is admin
Dim isAdminUser as Boolean = False

'Loop through the Persona array
For Each arrPersona as String in arrPersonas
    'Get the Persona object by name
    Dim aPersona as Persona = Loan.Session.Users.Personas.GetPersonaByName(arrPersona)
    'Check if current user has an administartor persona
    If currUser.Personas.Contains(aPersona) Then
        isAdminUser = True
        Exit For
    End If
Next

'Triggers print auto selection rule
Dim autoPrintField = loan.Fields("CX.AUTOPRINT.TEST")
autoPrintField.Value = "Y"

'Opens pipeline view and switches back to loan view, necessary for correct print dialog window
Macro.SendKeys("^+p")
Macro.SendKeys("^+l")

'Opens print dialog
Macro.SendKeys("^p")

'TAB keystrokes to "Add to eFolder" button
' 7 works for Admins
' 6 works for non admin users
Dim tabsToAddToEFolder = 0
If isAdminUser Then
    tabsToAddToEFolder = 7
Else
    tabsToAddToEFolder = 6
End If

'To Add to Efolder
Macro.SendKeys("{TAB " & tabsToAddToEFolder.ToString() & "}")

'Prints form
Macro.SendKeys("~")
'Forces SendKeys events to act
System.Windows.Forms.Application.DoEvents()

'Declare variable for current Encompass process
Dim p As System.Diagnostics.Process = System.Diagnostics.Process.GetCurrentProcess()

'Loop until File Manager is active window before trying to close it.
Do
    If p.MainWindowTitle = "File Manager"
        'Closes File Manager window
        Macro.SendKeys("%{F4}")
        'Forces SendKeys events to act
        System.Windows.Forms.Application.DoEvents()
        Exit Do
    End If
Loop Until p.MainWindowTitle = "File Manager"


'Disables print auto selection rule
autoPrintField.Value = "N"

'Declare name of print form expected to drop in Unassigned folder
'This should match the name of the document that is being printed
Dim attName = "Gift Letter"

'Declare variable for located attachment -> set to Nothing at first
Dim foundAtt As Attachment = Nothing

'Loop until attachment is done converting // loop stops checking if taking too long to convert
Try
    Do
        For Each att As Attachment In loan.Attachments
            If att.Title = attName AndAlso att.GetDocument() Is Nothing AndAlso att.Size > 0 Then
                foundAtt = att
                Exit Try
            End If
        Next
    Loop Until foundAtt IsNot Nothing
Catch
    MsgBox("Conversion of document took to long.", 16, "Problem")
    Exit Try
End Try

'Find or create eFolder document
Dim trackedDocs = Loan.Log.TrackedDocuments
Dim documentName = "eFolder Document Name"
Dim list = trackedDocs.GetDocumentsByTitle(documentName)
Dim trackedDoc = If(list.Count > 0, DirectCast(list(0), TrackedDocument), trackedDocs.Add(documentName, "Processing"))

Try
    'Move attachment to eFolder document
    trackedDoc.Attach(foundAtt)
    Macro.Alert(attName & " successfully added to eFolder.")
    Exit Try
Catch
    'Catch exception when Encompass tries to move the attachment but can't find it (still converting)
    Macro.Alert(attName & " is taking too long to convert. Go to Unassigned folder and manually move the form to its Document placeholder.")
End Try

    //Get the current user
User currUser = Loan.Session.GetCurrentUser();
//Create an array of Persona names
string[] arrPersonas = {"Super Administrator", "Administrator"};
//Create boolean to hold if user is admin
bool isAdminUser = false;

//Loop through the Persona array
foreach (string arrPersona in arrPersonas)
{
    Persona aPersona = Loan.Session.Users.Personas.GetPersonaByName(arrPersona);

    //Check if current user has persona and do something
    if (currUser.Personas.Contains(aPersona))
    {
        isAdminUser = true;
    }
}

//Trigger print auto select rule
Field autoPrintField = Loan.Fields["CX.AUTOPRINT.TEST"];
autoPrintField.Value = "Y";

//Opens pipeline view and switches back to loan view, necessary for correct print dialog window
Macro.SendKeys("^+p");
Macro.SendKeys("^+l");

//Opens print dialog
Macro.SendKeys("^p");

//TAB keystrokes to "Add to eFolder" button
// 7 works for Admins
// 6 works for non admin users
int tabsToAddToEFolder = 0;
if (isAdminUser)
{
    tabsToAddToEFolder = 7;
}
else
{
    tabsToAddToEFolder = 6;
}

//To Add to Efolder
Macro.SendKeys("{TAB " + tabsToAddToEFolder.ToString() + "}");

//Prints form
Macro.SendKeys("~");
//Forces SendKeys events to act
System.Windows.Forms.Application.DoEvents();

//Declare variable for current Encompass process
System.Diagnostics.Process p = System.Diagnostics.Process.GetCurrentProcess();

//Loop until File Manager is active window before trying to close it.
do
{
    if (p.MainWindowTitle == "File Manager")
    {
        //Closes File Manager window
        Macro.SendKeys("%{F4}");
        // Forces SendKeys events to act
        System.Windows.Forms.Application.DoEvents();
        break;
    }
} while (p.MainWindowTitle == "File Manager");


//Disables print auto selection rule
autoPrintField.Value = "N";

//Declare name of print form expected to drop in Unassigned folder
//This should match the name of the document that is being printed
string attName = "Gift Letter";

//Declare variable for located attachment -> set to Nothing at first
Attachment foundAtt = null;

//Loop until attachment is done converting // loop stops checking if taking too long to convert
try
{
    do
    {
        foreach (Attachment att in Loan.Attachments)
        {
            if (att.Title == attName && att.GetDocument() == null && att.Size > 0)
            {
                foundAtt = att;
                break;
            }
        }
    } while (foundAtt == null);
}   
catch
{
    System.Windows.Forms.MessageBox.Show("Conversion of document took to long.", "Problem", System.Windows.Forms.MessageBoxButtons.OK, System.Windows.Forms.MessageBoxIcon.Error);
}

//Find or create eFolder document
LogTrackedDocuments trackedDocs = Loan.Log.TrackedDocuments;
string documentName = "eFolder Document Name";
LogEntryList list = trackedDocs.GetDocumentsByTitle(documentName);
TrackedDocument trackedDoc = (list.Count > 0) ? (TrackedDocument)list[0] : trackedDocs.Add(documentName, "Processing");

try
{
    //Move attachment to eFolder document
    trackedDoc.Attach(foundAtt);
    Macro.Alert(attName + " successfully added to eFolder.");
}
catch
{
    //Catch exception when Encompass tries to move the attachment but can't find it (still converting)
    Macro.Alert(attName + " is taking too long to convert. Go to Unassigned folder and manually move the form to its Document placeholder.");
}
